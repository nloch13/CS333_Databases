DROP TABLE IF EXISTS EsportTeam;
DROP TABLE IF EXISTS VideoGame;
DROP TABLE IF EXISTS Gamer;
DROP TABLE IF EXISTS League;
DROP TABLE IF EXISTS Map;
DROP TABLE IF EXISTS Record_RD;


CREATE TABLE IF NOT EXISTS Record_RD
(
  RoundDifferential VARCHAR(30) NOT NULL,
  Place INT NOT NULL,
  PRIMARY KEY (Place)
);

CREATE TABLE IF NOT EXISTS Map
(
  MapName VARCHAR(30) NOT NULL,
  MapBanPersent INT NOT NULL,
  PRIMARY KEY (MapName)
);

CREATE TABLE IF NOT EXISTS League
(
  LeagueName VARCHAR(30) NOT NULL,
  Numofteams INT NOT NULL,
  PrizePool INT NOT NULL,
  Season VARCHAR(20) NOT NULL,
  PRIMARY KEY (LeagueName)
);

CREATE TABLE IF NOT EXISTS Gamer
(
  IGN VARCHAR(30) NOT NULL,
  Grade VARCHAR(20) NOT NULL,
  PlayerName VARCHAR(40) NOT NULL,
  Rank VARCHAR(20) NOT NULL,
  KDA DECIMAL(5, 2) NOT NULL,
  PRIMARY KEY (IGN)
);

CREATE TABLE IF NOT EXISTS VideoGame
(
  GameName VARCHAR(30) NOT NULL,
  System VARCHAR(30),
  ReleaseYear INT,
  IGN VARCHAR(30) NOT NULL,
  PRIMARY KEY (GameName),
  FOREIGN KEY (IGN) REFERENCES Gamer(IGN)
);

CREATE TABLE IF NOT EXISTS EsportTeam
(
  TeamRecord VARCHAR(5) NOT NULL,
  TeamName VARCHAR(30) NOT NULL,
  TeamAcronym VARCHAR(4) NOT NULL,
  GameName VARCHAR(30) NOT NULL,
  Place INT NOT NULL,
  LeagueName VARCHAR(30) NOT NULL,
  PRIMARY KEY (TeamName),
  FOREIGN KEY (GameName) REFERENCES VideoGame(GameName),
  FOREIGN KEY (Place) REFERENCES Record_RD(Place),
  FOREIGN KEY (LeagueName) REFERENCES League(LeagueName)
);
